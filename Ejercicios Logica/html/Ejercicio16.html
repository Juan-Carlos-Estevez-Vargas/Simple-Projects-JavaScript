const montoFinal = (monto = undefined, descuento = undefined) => {
   if (monto === undefined || descuento === undefined) return document.write("No ingresaste el monto o el descuento");
   if (typeof monto !== "number" || typeof descuento !== "number") return document.write("Los datos ingresados son incorrectos");

   return document.write( `Monto final: ${monto - (monto * descuento) / 100}` );
};

montoFinal(1000, 20);
